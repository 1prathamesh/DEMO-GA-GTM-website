<script>

fetch('/products')
.then(res => res.json())
.then(data => {
  data.forEach(p => {
    const btn = document.createElement("button");
    btn.innerText = "Add " + p.name;
    btn.onclick = () => addToCart(p);
    document.body.appendChild(btn);
  });
});

function addToCart(p){
  fetch('/cart',{
    method:"POST",
    headers:{'Content-Type':'application/json'},
    body:JSON.stringify(p)
  })
  .then(res=>res.json())
  .then(res=>{
    dataLayer.push(res.dataLayer);
    console.log(res.dataLayer);
  });
}

function purchase(){
  fetch('/purchase',{method:"POST"})
  .then(res=>res.json())
  .then(res=>{
    dataLayer.push(res.dataLayer);
  });
}
